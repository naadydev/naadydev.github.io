Examples: Data Root

You Can Define The Data Root 


                                   
                   < div class="myjtline" > </ div >
                
                

                
             $('.myjtline').jTLine({
                 callType: 'jsonObject',
                 structureObj: myMappedObjectB,
                 map: {
                    "dataRoot": "parent.ch2",
                    "title": "taskTitle",
                    "subTitle": "taskSubTitle",
                    "dateValue": "assignDate",
                    "pointCnt": "taskShortDate",
                    "bodyCnt": "taskDetails"
                 }
                });
